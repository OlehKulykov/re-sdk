67  static REVector3 crossProduct(const REVector3 & firstVector, const REVector3 & secondVector)
68  {
69 #if defined(__ARM_NEON__)

73  float32x4_t rzxy = { secondVector.z, secondVector.x, secondVector.y, 0.0f };
74  return REVector3(vmlsq_f32(vmulq_f32(*(float32x4_t *)&lyzx, *(float32x4_t *)&rzxy), *(float32x4_t *)&lzxy, *(float32x4_t *)&ryzx));
75 #else
76  return REVector3(( (firstVector.y * secondVector.z) - (firstVector.z * secondVector.y) ),
77  ( (firstVector.z * secondVector.x) - (firstVector.x * secondVector.z) ),
78  ( (firstVector.x * secondVector.y) - (firstVector.y * secondVector.x) ));
79 #endif

417  REVector3 & rotate(const REVector3 & arroundVector, const REFloat32 clockwiseRadianAngle)
418  {
419  const REFloat32 negativeHalfAngle = (-clockwiseRadianAngle / 2.0f);
420  const REFloat32 s = sinf(negativeHalfAngle);
421 
422  REVector3 normalizedVector(arroundVector);
423  normalizedVector.normalize();
424 
425  REFloat32 x = normalizedVector.x * s;
426  REFloat32 y = normalizedVector.y * s;
427  REFloat32 z = normalizedVector.z * s;
428 
429  const REFloat32 w = cosf(negativeHalfAngle);
430  const REFloat32 w2 = w * w;
431  const REFloat32 x2 = x * x;
432  const REFloat32 y2 = y * y;
433  const REFloat32 z2 = z * z;
434 
435  const REFloat32 resX = (x*(w2+x2-y2-z2)) + (y*2.0f*((x*y)+(w*z))) + (z*2.0f*((x*z)-(w*y)));
436  const REFloat32 resY = (x*2.0f*((x*y)-(w*z))) + (y*(w2-x2+y2-z2)) + (z*2.0f*((y*z)+(w*x)));
437  const REFloat32 resZ = (x*2.0f*((x*z)+(w*y))) + (y*2.0f*((y*z)-(w*x))) + (z*(w2-x2-y2+z2));
438 
439  x = resX;
440  y = resY;
441  z = resZ;
442 
443  return (*this);
444  }
