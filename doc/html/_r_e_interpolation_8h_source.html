336  static const REFloat32 bounceEaseIn(const REFloat32 p)
337  {
338  REFloat32 np = 1.0f - p;
339  if (np < (1.0f / 2.75f))
340  {
341  np = 7.5625f * np * np;
342  }
343  else if (np < (2.0f / 2.75f))
344  {
345  np = np - (1.5f / 2.75f);
346  np = 7.5625f * np * np + 0.75f;
347  }
348  else if (np < (2.5f / 2.75f))
349  {
350  np = np - (2.25f / 2.75f);
351  np = 7.5625f * np * np + 0.9375f;
352  }
353  else
354  {
355  np = np - (2.625f / 2.75f);
356  np = 7.5625f * np * np + 0.984375f;
357  }
358  return (1.0f - np);
359  }
