223  REVector2 & Divide(const REVector2 & anotherVector2D)
224  {
225  if ((anotherVector2D.x == 0.0f) || (anotherVector2D.y == 0.0f))
226  {
227  x = y = FLT_MAX;
228  }
229  else
230  {
231 #if defined(__ARM_NEON__)
232  float32x2_t *vLeft = (float32x2_t *)&armNeonVector;
233  float32x2_t *vRight = (float32x2_t *)&anotherVector2D.armNeonVector;
234  float32x2_t estimate = vrecpe_f32(*vRight);
235  estimate = vmul_f32(vrecps_f32(*vRight, estimate), estimate);
236  estimate = vmul_f32(vrecps_f32(*vRight, estimate), estimate);
237  armNeonVector = vmul_f32(*vLeft, estimate);
238 #else
239  x /= anotherVector2D.x;
240  y /= anotherVector2D.y;
241 #endif
242  }
243  return (*this);
244  }
