411  REVector3 & rotate(const REVector3 & arroundVector, const REFloat32 clockwiseRadianAngle)
412  {
413  const REFloat32 negativeHalfAngle = (-clockwiseRadianAngle / 2.0f);
414  const REFloat32 s = sinf(negativeHalfAngle);
415 
416  REVector3 normalizedVector(arroundVector);
417  normalizedVector.normalize();
418 
419  REFloat32 x = normalizedVector.x * s;
420  REFloat32 y = normalizedVector.y * s;
421  REFloat32 z = normalizedVector.z * s;
422 
423  const REFloat32 w = cosf(negativeHalfAngle);
424  const REFloat32 w2 = w * w;
425  const REFloat32 x2 = x * x;
426  const REFloat32 y2 = y * y;
427  const REFloat32 z2 = z * z;
428 
429  const REFloat32 resX = (x*(w2+x2-y2-z2)) + (y*2.0f*((x*y)+(w*z))) + (z*2.0f*((x*z)-(w*y)));
430  const REFloat32 resY = (x*2.0f*((x*y)-(w*z))) + (y*(w2-x2+y2-z2)) + (z*2.0f*((y*z)+(w*x)));
431  const REFloat32 resZ = (x*2.0f*((x*z)+(w*y))) + (y*2.0f*((y*z)-(w*x))) + (z*(w2-x2-y2+z2));
432 
433  x = resX;
434  y = resY;
435  z = resZ;
436 
437  return (*this);
438  }
