332  static const REFloat32 BounceEaseIn(const REFloat32 p)
333  {
334  REFloat32 np = 1.0f - p;
335  if (np < (1.0f / 2.75f))
336  {
337  np = 7.5625f * np * np;
338  }
339  else if (np < (2.0f / 2.75f))
340  {
341  np = np - (1.5f / 2.75f);
342  np = 7.5625f * np * np + 0.75f;
343  }
344  else if (np < (2.5f / 2.75f))
345  {
346  np = np - (2.25f / 2.75f);
347  np = 7.5625f * np * np + 0.9375f;
348  }
349  else
350  {
351  np = np - (2.625f / 2.75f);
352  np = 7.5625f * np * np + 0.984375f;
353  }
354  return (1.0f - np);
355  }
