214  {

216  {

222  float32x2_t *vLeft = (float32x2_t *)&armNeonVector;

224  float32x2_t estimate = vrecpe_f32(*vRight);
225  estimate = vmul_f32(vrecps_f32(*vRight, estimate), estimate);
226  estimate = vmul_f32(vrecps_f32(*vRight, estimate), estimate);
227  armNeonVector = vmul_f32(*vLeft, estimate);
228 #else
229  x /= anotherVector2D.x;

231 #endif
232  }
