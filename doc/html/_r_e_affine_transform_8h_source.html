79  REAffineTransform & concat(const REAffineTransform & t2)
80  {
81  REFloat32 arr1[6];
82  arr1[0] = (t2.a * a) + (t2.b * c);
83  arr1[1] = (t2.a * b) + (t2.b * d);
84  arr1[2] = (t2.c * a) + (t2.d * c);
85  arr1[3] = (t2.c * b) + (t2.d * d);
86  arr1[4] = (t2.tx * a) + (t2.ty * c) + tx;
87  arr1[5] = (t2.tx * b) + (t2.ty * d) + ty;
88  memcpy(arr, arr1, 6 * sizeof(REFloat32));
89  return (*this);
90  }
