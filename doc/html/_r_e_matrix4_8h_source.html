347  REMatrix4 & toLookAt(const REFloat32 eyeX,
348  const REFloat32 eyeY,
349  const REFloat32 eyeZ,
350  const REFloat32 centerX,

372  m01 = v.x;
373  m02 = n.x;
374 
375  m10 = u.y;
376  m11 = v.y;

407  static REMatrix4 createRotation(const REFloat32 radians, const REFloat32 x, const REFloat32 y, const REFloat32 z)
408  {
409  const REFloat32 cosV = cosf(radians);
410  const REFloat32 cospV = 1.0f - cosV;
411  const REFloat32 sinV = sinf(radians);
412  REVector3 n(x,y,z);
413  n.normalize();
414  REMatrix4 m;
415  m.line[0] = cosV + cospV * n.x * n.x;
416  m.line[1] = cospV * n.x * n.y + n.z * sinV;
417  m.line[2] = cospV * n.x * n.z - n.y * sinV;
418  m.line[4] = cospV * n.x * n.y - n.z * sinV;
419  m.line[5] = cosV + cospV * n.y * n.y;
420  m.line[6] = cospV * n.y * n.z + n.x * sinV;
421  m.line[7] = cospV * n.x * n.z + n.y * sinV;
422  m.line[8] = cospV * n.y * n.z - n.x * sinV;
423  m.line[10] = cosV + cospV * n.z * n.z;
424  m.line[15] = 1.0f;
425  return m;
426  }
