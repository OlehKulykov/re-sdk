381  static const REFloat32 bounceEaseIn(const REFloat32 p)
382  {
383  REFloat32 np = 1.0f - p;
384  if (np < (1.0f / 2.75f))
385  {
386  np = 7.5625f * np * np;
387  }
388  else if (np < (2.0f / 2.75f))
389  {
390  np = np - (1.5f / 2.75f);
391  np = 7.5625f * np * np + 0.75f;
392  }
393  else if (np < (2.5f / 2.75f))
394  {
395  np = np - (2.25f / 2.75f);
396  np = 7.5625f * np * np + 0.9375f;
397  }
398  else
399  {
400  np = np - (2.625f / 2.75f);
401  np = 7.5625f * np * np + 0.984375f;
402  }
403  return (1.0f - np);
404  }
