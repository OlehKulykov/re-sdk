51  const REFloat32 GetMagnitude() const
52  {
53 #if defined(__ARM_NEON__)
54  float32x4_t v = vmulq_f32(*(float32x4_t *)&armNeonQuaternion, *(float32x4_t *)&armNeonQuaternion);
55  float32x2_t v2 = vpadd_f32(vget_low_f32(v), vget_high_f32(v));
56  v2 = vpadd_f32(v2, v2);
57  return sqrtf(vget_lane_f32(v2, 0));
58 #else
59  return sqrtf((q[0] * q[0]) + (q[1] * q[1]) + (q[2] * q[2]) + (q[3] * q[3]));
60 #endif
61  }

77  REQuaternion GetConjugateQuaternion() const
78  {
79 #if defined(__ARM_NEON__)
80  float32x4_t *q = (float32x4_t *)&armNeonQuaternion;
81  uint32_t signBit = 0x80000000;

84  mask = vsetq_lane_u32(zeroBit, mask, 3);

91  REQuaternion GetInverseQuaternion() const
92  {
93 #if defined(__ARM_NEON__)
94  float32x4_t * q = (float32x4_t *)&armNeonQuaternion;
95  float32x4_t v = vmulq_f32(*q, *q);
96  float32x2_t v2 = vpadd_f32(vget_low_f32(v), vget_high_f32(v));
97  v2 = vpadd_f32(v2, v2);
98  float32_t scale = 1.0f / vget_lane_f32(v2, 0);
99  v = vmulq_f32(*q, vdupq_n_f32(scale));
100  uint32_t signBit = 0x80000000;
101  uint32_t zeroBit = 0x0;
102  uint32x4_t mask = vdupq_n_u32(signBit);
103  mask = vsetq_lane_u32(zeroBit, mask, 3);
104  return REQuaternion(vreinterpretq_f32_u32(veorq_u32(vreinterpretq_u32_f32(v), mask)));
105 #else
106  const REFloat32 scale = 1.0f / ((q[0] * q[0]) + (q[1] * q[1]) + (q[2] * q[2]) + (q[3] * q[3]));
107  return REQuaternion(q[0] * -scale, q[1] * -scale, q[2] * -scale, q[3] * scale);
108 #endif
109  }

125  REQuaternion & Multiply(const REQuaternion & aq)
126  {
127  const REFloat32 x1 = (q[3] * aq.q[0]) + (q[0] * aq.q[3]) + (q[1] * aq.q[2]) - (q[2] * aq.q[1]);
128  const REFloat32 y1 = (q[3] * aq.q[1]) + (q[1] * aq.q[3]) + (q[2] * aq.q[0]) - (q[0] * aq.q[2]);
129  const REFloat32 z1 = (q[3] * aq.q[2]) + (q[2] * aq.q[3]) + (q[0] * aq.q[1]) - (q[1] * aq.q[0]);
130  w = (q[3] * aq.q[3]) - (q[0] * aq.q[0]) - (q[1] * aq.q[1]) - (q[2] * aq.q[2]);
131  x = x1;
132  y = y1;
133  z = z1;
134  return (*this);
135  }
