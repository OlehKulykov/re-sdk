73  static size_t writeHeaders(void * buffer, size_t size, size_t nmemb, void * userp)
74  {
75  REURLConnectionCURLPrivate * connection = (REURLConnectionCURLPrivate *)userp;
76  const char * s = (const char *)buffer;
77  switch (*s)
78  {
79  case 'H':
80  {
81  float httpVersion = -1.0f;
82  int statusCode = -1;
83  if (sscanf(s, "HTTP/%f %i", &httpVersion, &statusCode) == 2)
84  {
85  connection->_mutableResponse->setStatusCode(statusCode);
86  connection->_mutableResponse->setHttpVersion(httpVersion);
87  }
88  }
89  break;
90  case 'C':
91  {
92  char contType[128] = { 0 };
93  unsigned long long contLen = 0;
94  if (sscanf(s, "Content-Type: %s", contType) == 1)
95  {
96  connection->_mutableResponse->setMIMEType(contType);
97  }
98  else if (sscanf(s, "Content-Length: %llu", &contLen) == 1)
99  {
100  connection->_mutableResponse->setExpectedContentSize((REUInt32)contLen);
101  connection->_mutableResponse->setIsExpectedContentSizeDetermined(true);
102  }
103  }
104  break;
105  case '\r':
106  {
107  if (strlen(s) == 2)
108  {
109  if (strncmp(s, "\r\n", 2) == 0)
110  {
111  connection->swapResponcesIfNeeded();
112  if (connection->_delegate)
113  {
114  connection->_delegate->URLConnectionReceivedResponse(connection, connection->_mutableResponse);
115  }
116  }
117  }
118  }
119  break;
120  default:
121  {
122 
123  }
124  break;
125  }
126  const size_t needWriteSize = size * nmemb;
127  return needWriteSize;
128  }
